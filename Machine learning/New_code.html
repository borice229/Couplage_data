# Dfinition de la fonction
Generate_output <- function(N, a, b, c, U0) {
  # Initialisation U0
  U <- numeric(N + 1)
  U[1] <- U0
  
  # Boucle pour calculer les valeurs de U_n
  for (n in 1:N) {
    U[n + 1] <- a * U[n]^c + b
  }
  
  return(U)
}

calculate_gradient <- function(N, a, b, c, U0) {
  h_values <- seq(0.001, 0.05, length.out = 10) # Dfinir une plage de h
  fU <- Generate_output(N, a, b, c, U0)
  
  # Matrices pour stocker les variations pour chaque h
  variations_a <- matrix(0, nrow = length(fU), ncol = length(h_values))
  variations_b <- matrix(0, nrow = length(fU), ncol = length(h_values))
  variations_c <- matrix(0, nrow = length(fU), ncol = length(h_values))
  
  # Calcul des variations pour chaque h
  for (i in seq_along(h_values)) {
    h <- h_values[i]
    variations_a[, i] <- (Generate_output(N, a + h, b, c, U0) - fU) / h
    variations_b[, i] <- (Generate_output(N, a, b + h, c, U0) - fU) / h
    variations_c[, i] <- (Generate_output(N, a, b, c + h, U0) - fU) / h
  }
  
  # Calcul de la moyenne des variations pour chaque paramtre
  gradient_a <- rowMeans(variations_a)
  gradient_b <- rowMeans(variations_b)
  gradient_c <- rowMeans(variations_c)
  
  # Combiner les rsultats en une matrice
  gradient_f <- round(cbind(gradient_a, gradient_b, gradient_c, U0),2)
  
  return(gradient_f)
}

calculate_sensitivity_indicateur <- function(gradients) {
  # Vrifier si la matrice des gradients est valide
  if (is.null(gradients) || nrow(gradients) == 0 || ncol(gradients) < 3) {
    stop("La matrice des gradients doit contenir au moins 3 colonnes et une ou plusieurs lignes.")
  }
  
  # Calculer la norme des gradients pour chaque ligne
  norms <- calculate_gradient_norm(gradients)
  
  # viter les divisions par zro en ajoutant une petite valeur epsilon
  epsilon <- 1e-6
  sensitivity_values <- rowSums(gradients[, 1:3]) / (norms + epsilon)
  
  # Calculer l'indicateur global en prenant la mdiane des sensibilits
  median_indicator <- round(median(sensitivity_values, na.rm = TRUE), 2)
  
  return(median_indicator)
}

get_sensibility <- function(stock) {
  # Calculer la mdiane de la colonne Indicator
  median_value <- median(stock$Indicator, na.rm = TRUE)
  
  # Trouver l'indice de la ligne dont la valeur d'Indicator est la plus proche de la mdiane
  closest_index <- which.min(abs(stock$Indicator - median_value))
  
  # Retourner la ligne correspondante
  selected_row <- stock[closest_index, ]
  
  return(selected_row)
}

1. Notion de bruit dans la collecte de donnes Le bruit en collecte de donnes dsigne les variations alatoires ou les informations non pertinentes qui perturbent les donnes recueillies. Ces perturbations ne refltent pas le phnomne rel mesur. Le bruit peut provenir de sources varies, telles que des erreurs de mesure, des variations environnementales ou des biais dans les instruments de collecte.

2. Problmes causs par le bruit La prsence de bruit peut entraner plusieurs problmes :

Baisse de la prcision : Le bruit rend les donnes moins fiables et rend difficile la distinction entre les variations alatoires et les tendances relles. Difficult interprter : Le bruit complique lanalyse, car il peut masquer les relations importantes entre les variables. Modles biaiss : En prsence de bruit important, les modles statistiques peuvent produire des rsultats inexacts ou non reprsentatifs.

3. Loi de probabilit du composant alatoire Le composant alatoire, cest--dire le bruit, peut souvent tre modlis par une loi normale (gaussienne), car les erreurs et perturbations alatoires ont tendance suivre une distribution symtrique autour de zro avec une variance spcifique. Cependant, dautres distributions peuvent tre utilises selon la nature du bruit (loi uniforme pour des erreurs constantes, loi exponentielle pour des erreurs asymtriques, etc.). La loi normale est souvent privilgie car elle reflte bien les erreurs alatoires dans de nombreux contextes.

2. Calcul de la prcision du modle sur le jeu de donnes dentranement

Le MAE sur le jeu de donnes dentranement se calcule laide de la formule suivante :

\[ MAE_{\text{train}} = \frac{1}{n} \sum_{i=1}^{n} |y_{\text{rel}, i} - y_{\text{prdit}, i}| \]

O : - \(y_{\text{rel}, i}\) reprsente la valeur relle de lobservation \(i\), - \(y_{\text{prdit}, i}\) est la valeur prdite pour lobservation \(i\), - \(n\) est le nombre total dobservations dans le jeu de donnes dentranement.

Le MAE permet de quantifier lajustement du modle aux donnes dentranement en indiquant la moyenne des erreurs absolues entre les valeurs relles et prdites. Un MAE faible indique une bonne prcision du modle, tandis quun MAE lev suggre que les prdictions du modle sont moins prcises.

# Fonction pour calculer le MAE pour chaque ligne
calculate_mae_per_row <- function(y_real_data, y_pred_data) {
  # Vrifier que les dimensions des matrices correspondent
  if (nrow(y_real_data) != nrow(y_pred_data)) {
    stop("Les matrices y_real_data et y_pred_data doivent avoir le mme nombre de lignes.")
  }
  
  # Calcul du MAE pour chaque ligne
  mae_per_row <- apply(cbind(y_real_data, y_pred_data), 1, function(row) {
    # Extraire la ligne de valeurs relles et prdites
    y_real <- row[1:ncol(y_real_data)]
    y_pred <- row[(ncol(y_real_data) + 1):length(row)]
    
    # Calcul de l'erreur absolue pour la ligne
    mae <- mean(abs(y_real - y_pred))
    return(mae)
  })
  
  # Calcul de la moyenne des MAE
  mae_global <- mean(mae_per_row)
  
  return(mae_global)
}

sgd <- function(y_observe_list, U0_list, N, learning_rate, max_iter) {
  # Initialisation des paramtres
  a <- 0  # Initialisation dans l'intervalle [2, 5]
  b <- 0  # Initialisation dans l'intervalle [1, 5]
  c <- 0  # Initialisation dans l'intervalle [0.1, 0.9]
  
  cost_history <- numeric(max_iter)  # Stockage des cots
  
  for (iter in 1:max_iter) {
    # Slection d'un exemple alatoire
    i <- sample(1:length(U0_list), 1)
    U0 <- U0_list[[i]]
    y_observe <- y_observe_list[i, ]
    
    # Calculer les prdictions
    y_pred <- Generate_output(N, a, b, c, U0)
    
    # Calcul des gradients
    grad_a <- sum(2 * (y_pred - y_observe) * U0^c) / length(y_observe)
    grad_b <- sum(2 * (y_pred - y_observe)) / length(y_observe)
    grad_c <- sum(2 * (y_pred - y_observe) * log(U0) * a * U0^c) / length(y_observe)
    
    # Mise  jour des paramtres avec contraintes
    a <- max(min(a - learning_rate * grad_a, max(params$a)), min(params$a))
    b <- max(min(b - learning_rate * grad_b, max(params$b)), min(params$b))
    c <- max(min(c - learning_rate * grad_c, max(params$c)), min(params$c))
  
    # Calcul du cot
    cost <- sum((y_pred - y_observe)^2) / length(y_observe)
    cost_history[iter] <- cost
    
    # Afficher le cot toutes les 100 itrations
    if (iter %% 100 == 0) {
      cat("Iteration:", iter, " Cost:", cost,

  }

4.1 Principle of the approach

1. Principe global du Machine Learning

Le Machine Learning (apprentissage automatique) consiste utiliser des algorithmes pour permettre un modle de sapprendre des donnes et damliorer ses performances sans tre explicitement programm. Lide est dentraner un modle avec des donnes dexemples (donnes dapprentissage) pour quil puisse ensuite faire des prdictions ou des classifications sur de nouvelles donnes. Il existe trois types principaux dapprentissage : supervis, non supervis et par renforcement.

2. Intrts et limites par rapport au couplage de modles de donnes

Intrts du Machine Learning : - Adaptabilit : Le Machine Learning peut sadapter des problmes complexes sans ncessiter une comprhension complte des relations sous-jacentes. - Automatisation : Il permet dautomatiser la prise de dcision en traitant de grandes quantits de donnes.

Limites par rapport au couplage de modles : - Interprtabilit limite : Les modles de Machine Learning, surtout les modles complexes comme les rseaux de neurones, sont souvent des botes noires, difficiles interprter. - Ncessit de donnes : Ces modles ncessitent de grandes quantits de donnes annotes pour entraner les modles efficacement. - Manque de connaissance physique : Le Machine Learning peut ne pas prendre en compte les lois physiques ou les connaissances spcifiques du domaine, contrairement au DataModel Coupling, qui modlise les relations physiques et peut tre plus prcis dans des contextes bien compris.

3. Fonctionnement dun algorithme de rseau de neurones

Un rseau de neurones est une structure inspire du cerveau humain qui comprend des neurones artificiels organiss en couches. Chaque neurone dans une couche prend des entres, effectue une transformation (souvent une somme pondre suivie dune activation), et transmet le rsultat la couche suivante. Lalgorithme ajuste les poids des connexions entre neurones pendant lentranement laide dune technique comme la descente de gradient, en cherchant minimiser lerreur entre les prdictions du modle et les valeurs relles.

4. Influence du nombre de neurones et de couches sur les rsultats

  • Nombre de neurones : Un nombre plus lev de neurones dans chaque couche permet au rseau de capturer des relations plus complexes dans les donnes, mais peut aussi mener un sur-apprentissage si le rseau devient trop complexe par rapport la quantit de donnes.
  • Nombre de couches : Un plus grand nombre de couches (rseau profond) permet au modle de capturer des relations de plus haut niveau et plus abstraites. Cependant, cela peut galement rendre lentranement plus difficile, ncessitant des techniques avances pour viter le problme de vanishing gradient et garantir une bonne gnralisation.

  TestB_S <- matrix(0, nrow = nrow(TestB), ncol = ncol(TestB))
